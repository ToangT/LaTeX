\paragraph*{\textbf{\large{Aufgabe 1}}}
	\begin{enumerate}[label=(\alph*)]
	\item Beweise mit Hilfe vollständiger Induktion  \\
	$\displaystyle\sum_{n=1}^{n} k = 1 + 2 + 3 + \cdots + n = \frac{n(n+1)}{2} $
	\end{enumerate}
	\begin{proof}
	  \textbf{Induktionsanfang:} Sei $n=1$. Dann gilt: \\
	  \begin{align*}
	    \displaystyle\sum_{n=1}^{1} n = 1  = \frac{1(1+1)}{2} 
	  \end{align*}
	  \textbf{Induktionsvoraussetzung:} Sei $n \in \N$ beliebig fix. Die zu beweisende
	                                    Aussage gelte für dieses $n$.\\
	  \textbf{Induktionsschritt:} Dann gilt diese Aussage auch für $n+1$ ,denn
	  \begin{align*}
	    \displaystyle\sum_{n=1}^{n+1} k = 1 + 2 + 3 + \cdots + n + (n+1) &= \frac{(n+1)(n+2)}{2} \\
	    1 + 2 + 3 + \cdots + n + (n+1) &= \frac{(n^2 + 3n + 2)}{2}	\\
	    1 + 2 + 3 + \cdots + n &= \frac{(n^2 + 3n + 2)}{2} - \frac{2(n+1)}{2} \\
	    1 + 2 + 3 + \cdots + n &= \frac{n^2 + n}{2} \\
	    1 + 2 + 3 + \cdots + n &= \frac{n(n+1)}{2}
	  \end{align*}
	\end{proof}
